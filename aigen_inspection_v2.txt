    22:     public Task<string> RenderStringAsync(
    23:         string templateText, TemplateContext ctx, CancellationToken ct = default)
    24:     {
    25:         var tpl = Template.Parse(templateText);
    26:         if (tpl.HasErrors)
    27:         {
    28:             var msgs = string.Join("\n", tpl.Messages.Select(m => m.ToString()));
    29:             throw new InvalidOperationException($"Error en plantilla Scriban:\n{msgs}");

    54:     private static string ToSnake(string name)
    55:     {
    56:         var sb = new System.Text.StringBuilder();
    57:         for (int i = 0; i < name.Length; i++)
    58:         {
    59:             if (char.IsUpper(name[i]) && i > 0) sb.Append('_');
    60:             sb.Append(char.ToLower(name[i]));
    61:         }
    62:         return sb.ToString();

    66: /// <summary>
    67: /// Funciones utilitarias para plantillas Scriban.
    68: /// Se importan via ScriptObject.Import(typeof(ScribanFunctions)).
    69: /// Scriban las expone en snake_case: Pascal->pascal, NullableCs->nullable_cs
    70: ///
    71: /// Uso en plantillas:
    72: ///   {{ pascal "mi_texto" }}   -> "MiTexto"
    73: ///   {{ camel  "mi_texto" }}   -> "miTexto"
    74: ///   {{ kebab  "MiTexto"  }}   -> "mi-texto"
    75: ///   {{ nullable_cs "int" true }} -> "int?"
    76: /// </summary>
    77: public static class ScribanFunctions
    78: {
    79:     public static string Pascal(string? s) =>
    80:         string.Concat((s ?? "").Split('_', ' ', '-')
    81:             .Select(w => w.Length > 0 ? char.ToUpper(w[0]) + w[1..] : w));
    82: 
    83:     public static string Camel(string? s)
    84:     {
    85:         var p = Pascal(s);
    86:         return p.Length > 0 ? char.ToLower(p[0]) + p[1..] : p;
    87:     }
    88: 
    89:     public static string Kebab(string? s) =>
    90:         System.Text.RegularExpressions.Regex
    91:             .Replace(s ?? "", "([A-Z])", "-$1")
    92:             .TrimStart('-')
    93:             .ToLower();
    94: 
    95:     public static string Upper(string? s) => (s ?? "").ToUpperInvariant();
    96: 
    97:     public static string Lower(string? s) => (s ?? "").ToLowerInvariant();
    98: 
    99:     public static string NullableCs(string? type, object? nullable)
   100:     {
   101:         // Scriban puede pasar bool como bool, int (0/1) o string ("true"/"false")
   102:         var isNullable = nullable switch
   103:         {
   104:             bool b   => b,
   105:             int  i   => i != 0,
   106:             string s => s.Equals("true", StringComparison.OrdinalIgnoreCase),
   107:             _        => false
   108:         };
   109:         return isNullable && type != "string" ? (type ?? "") + "?" : type ?? "";
   110:     }
